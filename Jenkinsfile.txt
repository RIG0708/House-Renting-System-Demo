pipeline {
     agent any
     stages {
         stage('Restore') {
             steps {
                 bat 'dotnet restore HouseRentingSystem.sln'
             }
          }
         stage('Build') {
             step {
                bat 'dotnet build HouseRentingSystem.sln --configuration Release'
             }
          }
          stage('Test') {
              step {
                 bat 'dotnet test HouseRentingSystem.Tests/HouseRentingSystem.Tests.csproj'
                 }
          }
      }
}
